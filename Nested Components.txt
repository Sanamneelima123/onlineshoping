Nested Components (Means one A component data show to B component)

import React from "react";

const Acomponent = () =>{
  return <h1>this is A component data</h1>;
}

const Bcomponent = () =>{
  return <Acomponent />;
}

const App = () =>{
  return(
     <div>
        <Bcomponent />
     </div>
  )
}

export default App;
